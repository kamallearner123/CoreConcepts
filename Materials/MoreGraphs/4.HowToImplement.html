  • Adjacency List - Efficient for storing sparse graphs.
  • Adjacency Matrix - Good for dense graphs.
  • Edge List - Easy to implement but inefficient for searches.

3Ô∏è‚É£ Adjacency List Implementation

class Graph:
    def __init__(self):
        self.graph = {}

    def add_edge(self, u, v):
        if u not in self.graph:
            self.graph[u] = []
        self.graph[u].append(v)

    def display(self):
        for node in self.graph:
            print(node, "->", self.graph[node])

g = Graph()
g.add_edge(0, 1)
g.add_edge(0, 2)
g.add_edge(1, 2)
g.display()

4Ô∏è‚É£ Adjacency Matrix Implementation

import numpy as np
class GraphMatrix:
    def __init__(self, size):
        self.matrix = np.zeros((size, size), dtype=int)

    def add_edge(self, u, v):
        self.matrix[u][v] = 1

    def display(self):
        print(self.matrix)

g = GraphMatrix(3)
g.add_edge(0, 1)
g.add_edge(1, 2)
g.display()

5Ô∏è‚É£ Graph Traversal Algorithms

üîπ Depth-First Search (DFS)

def dfs(graph, node, visited=set()):
    if node not in visited:
        print(node, end=" ")
        visited.add(node)
        for neighbor in graph[node]:
            dfs(graph, neighbor, visited)

graph = {0: [1, 2], 1: [2], 2: [3], 3: []}
dfs(graph, 0)

üîπ Breadth-First Search (BFS)

from collections import deque
def bfs(graph, start):
    queue = deque([start])
    visited = set([start])

    while queue:
        node = queue.popleft()
        print(node, end=" ")
        for neighbor in graph[node]:
            if neighbor not in visited:
                queue.append(neighbor)
                visited.add(neighbor)

graph = {0: [1, 2], 1: [2], 2: [3], 3: []}
bfs(graph, 0)

6Ô∏è‚É£ Shortest Path Algorithms

üîπ Dijkstra's Algorithm (Greedy Approach)

import heapq
def dijkstra(graph, start):
    heap = [(0, start)]
    distances = {node: float('inf') for node in graph}
    distances[start] = 0

    while heap:
        (cost, node) = heapq.heappop(heap)
        for neighbor, weight in graph[node]:
            new_cost = cost + weight
            if new_cost < distances[neighbor]:
                distances[neighbor] = new_cost
                heapq.heappush(heap, (new_cost, neighbor))

    return distances

graph = {0: [(1, 4), (2, 1)], 1: [(3, 1)], 2: [(1, 2), (3, 5)], 3: []}
print(dijkstra(graph, 0))

üîπ Floyd-Warshall Algorithm (Dynamic Programming)

def floyd_warshall(graph, num_vertices):
    dist = [[float('inf')] * num_vertices for _ in range(num_vertices)]
    
    for u in graph:
        for v, w in graph[u]:
            dist[u][v] = w
    for i in range(num_vertices):
        dist[i][i] = 0

    for k in range(num_vertices):
        for i in range(num_vertices):
            for j in range(num_vertices):
                dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j])
    
    return dist

graph = {0: [(1, 3), (2, 6)], 1: [(2, 2), (3, 1)], 2: [(3, 4)], 3: []}
print(floyd_warshall(graph, 4))
